@pytest.mark.asyncio
async def test_request_body_rewritten_with_routed_model():
    """handle_request rewrites body model when downgrade is triggered."""
    import httpx
    from unittest.mock import AsyncMock, patch, MagicMock

    cfg = _cfg(team_budgets={"eng": 100.0}, threshold_pct=20.0)
    body = json.dumps({"model": "gpt-4o", "messages": [{"role": "user", "content": "hi"}]}).encode()

    # Mock provider
    mock_provider = MagicMock()
    mock_provider.name = "openai"
    mock_provider.base_url = "https://api.openai.com"

    # Mock the HTTP response from upstream
    mock_response = MagicMock()
    mock_response.status_code = 200
    mock_response.headers = {"content-type": "application/json"}
    mock_response.content = json.dumps({
        "id": "chatcmpl-test",
        "model": "gpt-4o-mini",
        "choices": [{"message": {"role": "assistant", "content": "Hello"}, "finish_reason": "stop"}],
        "usage": {"prompt_tokens": 10, "completion_tokens": 5, "total_tokens": 15},
    }).encode()

    captured_requests = []

    async def mock_request(*args, **kwargs):
        captured_requests.append(kwargs.get("content", b""))
        return mock_response

    with patch(TEAM_SPEND_PATCH, new_callable=AsyncMock, return_value={"eng": 85.0}):
        with patch("burnlens.storage.database.insert_request", new_callable=AsyncMock, return_value=1):
            with patch("burnlens.storage.database.init_db", new_callable=AsyncMock):
                mock_client = AsyncMock(spec=httpx.AsyncClient)
                mock_client.request = mock_request

                tags_header = {"x-burnlens-tag-team": "eng"}
                from burnlens.proxy.interceptor import handle_request
                status, _, resp_body, _ = await handle_request(
                    client=mock_client,
                    provider=mock_provider,
                    path="/v1/chat/completions",
                    method="POST",
                    headers=tags_header,
                    body_bytes=body,
                    query_string="",
                    db_path=":memory:",
                    config=cfg,
                )

    # Unconditional assertion — exactly 1 request must have been captured
    assert len(captured_requests) == 1, f"Expected 1 request captured, got {len(captured_requests)}"
    # The body sent to upstream must contain the routed model — unconditional
    sent_body = json.loads(captured_requests[0])
    assert sent_body["model"] == "gpt-4o-mini", f"Expected gpt-4o-mini, got {sent_body.get('model')}"
