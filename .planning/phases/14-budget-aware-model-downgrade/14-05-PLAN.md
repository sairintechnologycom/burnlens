The endpoint computes real savings using calculate_cost() on the original model — NOT hardcoded 0.0.

```python
@router.get("/routing-stats")
async def routing_stats(request: Request) -> dict:
    """Return downgrade routing statistics for today and the current month."""
    from datetime import datetime, timezone
    from burnlens.cost.calculator import calculate_cost, TokenUsage

    db_path = _db_path(request)
    today = datetime.now(timezone.utc).strftime("%Y-%m-%d")
    month_start = datetime.now(timezone.utc).strftime("%Y-%m-01")

    async with aiosqlite.connect(db_path) as db:
        db.row_factory = aiosqlite.Row

        # Today's downgrade count
        cursor = await db.execute(
            "SELECT COUNT(*) FROM requests WHERE downgrade_reason IS NOT NULL AND DATE(timestamp) = ?",
            (today,),
        )
        row = await cursor.fetchone()
        downgrades_today = row[0] if row else 0

        # This month's downgrade count
        cursor = await db.execute(
            "SELECT COUNT(*) FROM requests WHERE downgrade_reason IS NOT NULL AND timestamp >= ?",
            (month_start,),
        )
        row = await cursor.fetchone()
        downgrades_month = row[0] if row else 0

        # Savings today: sum of (original_model cost - actual cost) for downgraded rows
        cursor = await db.execute(
            """SELECT provider, model, cost_usd, input_tokens, output_tokens,
                      reasoning_tokens, cache_read_tokens, cache_write_tokens
               FROM requests
               WHERE downgrade_reason IS NOT NULL AND DATE(timestamp) = ?""",
            (today,),
        )
        rows_today = await cursor.fetchall()

        # Savings this month
        cursor = await db.execute(
            """SELECT provider, model, cost_usd, input_tokens, output_tokens,
                      reasoning_tokens, cache_read_tokens, cache_write_tokens
               FROM requests
               WHERE downgrade_reason IS NOT NULL AND timestamp >= ?""",
            (month_start,),
        )
        rows_month = await cursor.fetchall()

    def _compute_savings(rows) -> float:
        total = 0.0
        for r in rows:
            try:
                usage = TokenUsage(
                    input_tokens=r["input_tokens"] or 0,
                    output_tokens=r["output_tokens"] or 0,
                    reasoning_tokens=r["reasoning_tokens"] or 0,
                    cache_read_tokens=r["cache_read_tokens"] or 0,
                    cache_write_tokens=r["cache_write_tokens"] or 0,
                )
                original_cost = calculate_cost(r["provider"], r["model"], usage)
                total += max(0.0, original_cost - (r["cost_usd"] or 0.0))
            except Exception:
                pass  # unknown model pricing — skip row, don't crash
        return round(total, 6)

    return {
        "downgrades_today": downgrades_today,
        "saved_usd_today": _compute_savings(rows_today),
        "downgrades_this_month": downgrades_month,
        "saved_usd_this_month": _compute_savings(rows_month),
    }
```
