```python
@app.command()
def routing(
    config: Optional[Path] = typer.Option(None, "--config", "-c"),
    today_only: bool = typer.Option(False, "--today", help="Show today's downgrade events only"),
    json_output: bool = typer.Option(False, "--json", help="Machine-readable JSON output"),
) -> None:
    """Show downgrade routing activity.

    Displays all requests where BurnLens silently routed to a cheaper model
    due to budget threshold. Use --today to filter to today only, --json for
    machine-readable output.
    """
    import json as json_mod
    import aiosqlite
    from datetime import date

    cfg = load_config(config)

    async def _run() -> None:
        where_clause = "downgrade_reason IS NOT NULL"
        params: list = []
        if today_only:
            where_clause += " AND DATE(timestamp) = ?"
            params.append(date.today().isoformat())

        async with aiosqlite.connect(cfg.db_path) as db:
            db.row_factory = aiosqlite.Row
            cursor = await db.execute(
                f"SELECT timestamp, model, routed_model, downgrade_reason, tags "
                f"FROM requests WHERE {where_clause} ORDER BY timestamp DESC LIMIT 200",
                params,
            )
            rows = await cursor.fetchall()

        results = [dict(r) for r in rows]

        if json_output:
            console.print(json_mod.dumps(results, indent=2))
            return

        if not results:
            console.print("[yellow]No downgrade events found.[/yellow]")
            return

        from rich.table import Table
        table = Table(title="[bold]Routing Downgrades[/bold]", expand=True)
        table.add_column("Timestamp", style="dim")
        table.add_column("Original Model", style="cyan")
        table.add_column("Routed Model", style="green")
        table.add_column("Reason")
        # Budget Left is not stored in DB — only kept in memory during decide_route().
        # Future: store budget_remaining_usd/pct in RequestRecord for this column.
        table.add_column("Budget Left")

        for r in results:
            table.add_row(
                r["timestamp"][:19] if r["timestamp"] else "-",
                r["model"] or "-",
                r["routed_model"] or "-",
                r["downgrade_reason"] or "-",
                "-",
            )
        console.print(table)

    asyncio.run(_run())
```
